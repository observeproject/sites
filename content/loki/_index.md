### loki_alerts

##### LokiRequestErrors

{{< code lang="yaml" >}}
alert: LokiRequestErrors
annotations:
  message: |
    {{ $labels.job }} {{ $labels.route }} is experiencing {{ printf "%.2f" $value }}% errors.
expr: |
  100 * sum(rate(loki_request_duration_seconds_count{status_code=~"5.."}[2m])) by (namespace, job, route)
    /
  sum(rate(loki_request_duration_seconds_count[2m])) by (namespace, job, route)
    > 10
for: 15m
labels:
  severity: critical
{{< /code >}}
 
##### LokiRequestPanics

{{< code lang="yaml" >}}
alert: LokiRequestPanics
annotations:
  message: |
    {{ $labels.job }} is experiencing {{ printf "%.2f" $value }}% increase of panics.
expr: |
  sum(increase(loki_panic_total[10m])) by (namespace, job) > 0
labels:
  severity: critical
{{< /code >}}
 
##### LokiRequestLatency

{{< code lang="yaml" >}}
alert: LokiRequestLatency
annotations:
  message: |
    {{ $labels.job }} {{ $labels.route }} is experiencing {{ printf "%.2f" $value }}s 99th percentile latency.
expr: |
  namespace_job_route:loki_request_duration_seconds:99quantile{route!~"(?i).*tail.*|/schedulerpb.SchedulerForQuerier/QuerierLoop"} > 1
for: 15m
labels:
  severity: critical
{{< /code >}}
 
##### LokiTooManyCompactorsRunning

{{< code lang="yaml" >}}
alert: LokiTooManyCompactorsRunning
annotations:
  message: |
    {{ $labels.cluster }} {{ $labels.namespace }} has had {{ printf "%.0f" $value }} compactors running for more than 5m. Only one compactor should run at a time.
expr: |
  sum(loki_boltdb_shipper_compactor_running) by (namespace, cluster) > 1
for: 5m
labels:
  severity: warning
{{< /code >}}
